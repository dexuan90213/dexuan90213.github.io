Mac - Defaults

2020-03-25 1 min read # Defaults # Mac # macOS

鍵盤按鍵按住時重複輸入

# Enable Key Repeat
defaults write -g ApplePressAndHoldEnabled -bool false

# Disable Key Repeat
defaults write -g ApplePressAndHoldEnabled -bool true

加速鍵盤按鍵重複輸入速度

defaults write NSGlobalDomain KeyRepeat -int 1
