Domain-Specific Language (DSL)

某個領域的特定語言

RDBMS 關聯式資料庫

可以從 A 資料庫查到 B 資料庫的資料，互相可已關聯起來

建立資料庫

create database `super_store`;

使用資料庫

use super_store;

刪除資料庫

drop database `super_store`;

table

PK = 主鍵
NN = NotNull 不能為空
UQ = Uniq 唯一

追加欄位

ALTER TABLE `heroes`
ADD COLUMN super_power VARCHAR(100);

刪除欄位

ALTER TABLE `heroes`
DROP COLUMN super_power;

刪除資料表

DROP TABLE `heroes`

新增資料

INSERT INTO heroes
(name, genderm age, hero_level, gero_rank, description) VALUES ()

查詢全部資料

SELECT *
FROM heroes;

查詢所有 S 級英雄資料

SELECT *
FROM heroes
WHERE hero_level = 'S'

# Rails ORM
Hero.select(:name, :gender).where(hero_level: 'S', gender: 'F')

NULL 判斷

SELECT *
FROM heroes
WHERE age IS NULL;

# Rails ORM
Hero.where(age: nil)

like

SELECT *
FROM heroes
WHERE name LIKE '%背心%';

% 為萬用字元，前面的拿掉就是 背心...開頭

# Rails ORM
Hero.where("name like '%背心%'")

區間查詢

SELECT *
FROM heroes
WHERE age >= 10 and age <= 25;

SELECT *
FROM heroes
WHERE age BETWEEN 10 AND 25;

Hero.where(age: 10..25)

SELECT *
FROM heroes
WHERE age >= 10;

Hero.where('age >= 10') # 加字串

OR

SELECT *
FROM heroes
WHERE hero_level = 'S' or hero_level = 'A';

SELECT *
FROM heroes
WHERE hero_level IN('S', 'A');

不等於

SELECT *
FROM heroes
WHERE hero_level <> 'S';

SELECT *
FROM heroes
WHERE hero_level != 'S';

更新資料 UPDATE

UPDATE heroes
SET age = 10
WHERE id = 25

大量更新資料 Safe_mode

UPDATE heroes
SET age = age + 1;
-- 會壞掉
-- Error Code: 1175. You are using safe update mode and you tried to update a table without a WHERE that uses a KEY column.  To disable safe mode, toggle the option in Preferences -> SQL Editor and reconnect.

set sql_safe_updates = 0;
UPDATE heroes
SET age = age + 1;

刪除資料庫

set sql_safe_updates = 0;

delete from heroes
where hero_level = 'C';

select * from heroes;

計算數量 count

select count(*)
from heroes
where hero_level = 'A';
-- 只會顯示 18

Hero.where(hero_level: 'A').count
Hero.count(hero_level: 'A')

加總、平均

select sum(age)
from heroes
where hero_level = 'A' AND age IS NOT NULL;

select avg(age)
from heroes
where hero_level = 'A';

最大 最小

select min(age)
from heroes
where hero_level = 'A';

select max(age)
from heroes
where hero_level = 'A';

分組 GROUP

select hero_level as '英雄等級',
 avg(age) as '平均年齡', min(age), max(age)
from heroes
group by hero_level;

剔除重複的資料

select distinct danger_level
 from monsters;

排序

-- 反向排序 age 由大到小
select *
from heroes
where hero_level = 'S'
order by age DESC;

# Rails ORM
Hero.where(hero_level: 'S').order('age DESC')
Hero.where(hero_level: 'S').order(age: :desc)

限制數量

-- 反向排序 age 由大到小
select *
from heroes
where hero_level = 'S'
order by age DESC
LIMIT 3;

# Rails ORM
Hero.where(hero_level: 'S').order(age: :desc).limit(3)

交集

-- inner join
SELECT * FROM t1
join t2
on t1.username = t2.name

-- left join
SELECT * FROM t1
left join t2
on t1.username = t2.name

-- right join
SELECT * FROM t1
right join t2
on t1.username = t2.name
